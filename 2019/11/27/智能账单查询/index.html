– 账单查询
SELECT ACCT_ITEM_ID,
PROD_INST_ID,
AMOUNT 10000.0 as AMOUNT,
BALANCE_PAID 10000.0 as BALANCE_PAID,
PRICING_PLAN_ID,
REPOSITORY_ID,
ACCT_ITEM_TYPE_ID,
SERVICE_ID,
ITEM_SOURCE_ID,
STATUS_CD,
ACCT_ID,
SERV_PRODUCT_ID
FROM ACCT_ITEM_23
WHERE ACCT_ID = 1080066917
AND BILLING_CYCLE_ID = 314
AND ACCT_ITEM_TYPE_ID NOT IN (SELECT ACCT_ITEM_TYPE_ID
FROM ACCT_ITEM_23
WHERE ACCT_ID = 1080066917
AND BILLING_CYCLE_ID = 314
GROUP BY ACCT_ITEM_TYPE_ID
HAVING SUM(AMOUNT) = 0);

找到对应的offer_inst_id（最好通过PRICING_PLAN_ID）
select distinct OFFER_INST_ID from ACCT_ITEM_23 where ACCT_ITEM_ID in (268717884391) order by OFFER_INST_ID desc ;

通过offer_inst_id找到礼包实例id
select * from OFFER_PACK_INSTANCE_DETAIL where OFFER_INST_ID=122795175239;

通过上面找到的礼包实例id ，查询礼包下面有多少销售品实例
select offer_inst_id from OFFER_PACK_INSTANCE_DETAIL where PACK_OFFER_INST_ID=46778;

去 OFFER_OBJ_INST_REL 找到对应的instance_id（prod_inst_id）
SELECT distinct instance_id FROM OFFER_OBJ_INST_REL WHERE OBJ_TYPE = ‘100000’ AND EFF_DATE <= to_date(‘2019-07-31 23:59:59’,’yyyy-mm-dd hh24:mi:ss’) AND EXP_DATE >= to_date(‘2019-07-31 23:59:59’,’yyyy-mm-dd hh24:mi:ss’) AND OFFER_INST_ID in (122795175239,122795175294,122795175286,122795175277)

上面查出来的prod_inst_id,在账单表里面对应的记录，也归到礼包中去

礼包下面直接挂的销售品实例对应的账单数据，以及通过prod_inst_id 找到的账单数据，就是最终大礼包的数据
